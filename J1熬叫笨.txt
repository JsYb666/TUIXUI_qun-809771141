-- 加载库
local Lib = loadstring(game:HttpGet("https://raw.githubusercontent.com/Hosvile/Refinement/main/InfinitiveUI", true))()

-- 创建窗口
local Win = Lib:CreateWindow("ah", 1, nil, nil)

-- 创建一个通用的标签函数
local function createTab(tabName, tabIndex)
    local Tab, name = Win:CreateTab(tabName, function() warn(tabIndex) end)

    Tab:CreateButton("指令", function()
        loadstring(game:HttpGet("https://raw.githubusercontent.com/EdgeIY/infiniteyield/master/source", true))()
    end)
    
    Tab:CreateButton("控制台", function()
        game:GetService("StarterGui"):SetCore("DevConsoleVisible", true)
    end)

    for i = 1, 1 do
        Tab:CreateButton("点击 " .. i, function()
            print(i, name)
        end)

        Tab:CreateToggle("开关 " .. i, false, function(t)
            print(i, name, t)
        end)

        local max = math.random(500, 5000)
        Tab:CreateSlider("滑块 " .. i, 50, max, max / math.random(2, 6), function(v)
            print(i, name, v)
        end)

        Tab:CreateDropdown("下拉 " .. i, {
            {"命名", {}}, "1", "2", "3", "4", ""
        }, false, function(c, f)
            print(i, name, c, f)
        end)

        local Textbox = Tab:CreateTextbox("输入 " .. i, "FFlag")
        Textbox:GetPropertyChangedSignal("Text"):Connect(function()
            print(Textbox.Text)
        end)
    end
end

-- 创建标签
createTab("标签 1", 1)
createTab("标签 2", 2)
